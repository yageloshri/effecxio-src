שיחה
בנה לי REST API עם Express.js לניהול משתמשים
הנה קוד מוכן עם Express.js. הוספתי טיפול בשגיאות, pagination ו-validation.
תוסיף גם אימות עם JWT
הוספתי middleware לאימות. שימו לב שצריך להגדיר JWT_SECRET כמשתנה סביבה.
פלט קוד 
const express = require('express');
const jwt = require('jsonwebtoken');

// JWT auth middleware
const auth = (req, res, next) => {
  const token = req.headers
    .authorization
    ?.split(' ')[1];
  if (!token)
    return res
      .status(401)
      .json({error:'Unauthorized'});
  try {
    req.user = jwt.verify(
      token,
      process.env.JWT_SECRET
    );
    next();
  } catch {
    res.status(403)
      .json({error:'Invalid'});
  }
};

app.get('/api/users',
  auth,
  async (req, res) => {
    const users = await User
      .find()
      .limit(20);
    res.json({data:users});
  }
);
